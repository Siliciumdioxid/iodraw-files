flowchart TB
    %% 定义节点样式
    classDef input fill:#e1f5ff,stroke:#0078d4,stroke-width:2px;
    classDef process fill:#f3f3f3,stroke:#666,stroke-width:1px;
    classDef branch fill:#fff4cc,stroke:#ff9800,stroke-width:2px;
    classDef output fill:#e8f5e9,stroke:#4caf50,stroke-width:2px;

    %% 1. 输入层
    Input[/"原始3D激光点云\n（带运动畸变）"/]:::input

    %% 2. 特征提取层
    Feature[特征提取\n曲率计算：c=1/|S|·∥X∥ ∥∑(X−Xj)∥\n→ 边缘点（c最大）+ 平面点（c最小）]:::process

    %% 3. 双线程并行分支
    subgraph Parallel [双线程并行处理]
        direction TB
        %% 左分支：Lidar Odometry（10Hz）
        subgraph Odo [Lidar Odometry（10Hz 高频）]
            Odo1[特征匹配\n点→线 / 点→面]:::process
            Odo2[LM优化\n求解6DOF位姿]:::process
            Odo3[运动畸变补偿\n生成无畸变点云]:::process
            Odo1 --> Odo2 --> Odo3
        end

        %% 右分支：Lidar Mapping（1Hz）
        subgraph Map [Lidar Mapping（1Hz 低频）]
            Map1[无畸变点云投影\n→ 世界坐标系]:::process
            Map2[与全局地图精配准\n几何特征+迭代优化]:::process
            Map3[更新全局位姿 + 优化地图]:::process
            Map1 --> Map2 --> Map3
        end
    end

    %% 4. 位姿融合层
    Fusion[位姿融合\n全局位姿（Map）⊕ 帧间增量（Odo）\n时间线性插值：T(t) = T_k^W + (t-tk)/(t-tk+1)·T_{k+1}^L]:::branch

    %% 5. 输出层
    Output[/"输出\n→ 10Hz 高频高精度位姿\n→ 全局一致点云地图"/]:::output

    %% 连接逻辑
    Input --> Feature
    Feature --> Parallel
    Odo3 --> Fusion
    Map3 --> Fusion
    Fusion --> Output

    %% 样式绑定
    class Input,Output input;
    class Feature,Odo1,Odo2,Odo3,Map1,Map2,Map3 process;
    class Fusion branch;
